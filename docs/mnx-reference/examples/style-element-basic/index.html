Styling elements (basic)

This document contains a <style> element that effectively says "color every <note> element blue." Because the <style> element lives within <global>, the rule applies to the entire document.

{ "mnx": { "version": 1 }, "global": { "measures": [ { "barline": { "type": "regular" }, "time": { "count": 4, "unit": 4 } } ], "styles": [ { "color": "#0080FF", "selector": "note" } ] }, "parts": [ { "measures": [ { "clefs": [ { "clef": { "position": -2, "sign": "G" } } ], "sequences": [ { "content": [ { "type": "event", "duration": { "base": "quarter" }, "notes": [ { "pitch": { "octave": 5, "step": "C" } } ] }, { "type": "event", "duration": { "base": "quarter" }, "notes": [ { "pitch": { "octave": 5, "step": "E" } } ] }, { "type": "event", "duration": { "base": "quarter" }, "notes": [ { "pitch": { "octave": 5, "step": "G" } } ] }, { "type": "event", "duration": { "base": "quarter" }, "notes": [ { "pitch": { "octave": 6, "step": "C" } } ] } ] } ] } ] } ] }
